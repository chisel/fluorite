function generateRandomString(length) {

  const charset = 'qwertyuiopasdfghjklzxcvbnm1234567890';
  let string = '';

  for ( let i = 0; i < length; i++ ) {

    string += charset.charAt(Math.floor(Math.random() * charset.length));

  }

  return string;

}
